
model {
  # priors
  beta ~ dnorm(0,0.0001)
  sigma ~ dunif(0,10)
  tau <- pow(sigma,-2)
  for (i in 1:S) {
    a[i] ~ dnorm(0,0.0001)
  }
  # model of linear trend in abundance
  for (i in 1:S) {
    r[i] ~ dnorm(beta, tau)
    for (j in 1:T) {
      y[i,j] ~ dpois(mu[i,j])
      log(mu[i,j]) <- a[i] + r[i]*t[j]
    }
  }
}
